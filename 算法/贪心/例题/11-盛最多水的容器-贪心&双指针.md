```python
class Solution:
    def maxArea(self, height: List[int]) -> int:
        left = 0
        right = len(height) - 1
        ans = float("-inf")
        while left < right:
            vol = min(height[left], height[right]) * (right - left)
            ans = max(ans, vol)
            if height[left] <= height[right]:
                left += 1
            elif height[left] > height[right]:
                right -= 1
        return ans
```
