class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        direction = ((0,1), (1,0), (0,-1), (-1,0))

        def dfs(row, col):
            grid[row][col] = "0"
            for dr, dc in direction:
                cur_r = row + dr
                cur_c = col + dc
                if 0 <= cur_r < rows and 0 <= cur_c < cols and grid[cur_r][cur_c] == "1":
                    dfs(cur_r, cur_c)

        def bfs(row, col):
            queue = [(row, col)]
            while queue:
                r, c = queue.pop(0)
                for dr, dc in direction:
                    cur_r = r + dr
                    cur_c = c + dc
                    if 0<=cur_r<rows and 0<=cur_c<cols and grid[cur_r][cur_c] == "1":
                        grid[cur_r][cur_c] = "0"
                        queue.append((cur_r, cur_c))

        rows = len(grid)
        cols = len(grid[0])
        cnt = 0
        for r in range(rows):
            for c in range(cols):
                if grid[r][c] == "1":
                    bfs(r, c)
                    cnt += 1
        return cnt
