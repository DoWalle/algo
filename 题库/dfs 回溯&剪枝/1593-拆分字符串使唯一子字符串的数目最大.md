class Solution:
    def maxUniqueSplit(self, s: str) -> int:
        def dfs(path, begin):

            if size - begin + len(path) <= self.ans:
                return

            if begin == size:
                self.ans = max(self.ans, len(path))
                return

            for i in range(begin, size):
                tmp = s[begin:i+1]
                if tmp not in path:
                    path.add(tmp)
                    dfs(path, i + 1)
                    path.remove(tmp)

        size = len(s)
        path = set()
        self.ans = 0
        dfs(path, 0)

        return self.ans
