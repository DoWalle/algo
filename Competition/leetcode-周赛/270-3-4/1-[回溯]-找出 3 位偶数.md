#### [2094. 找出 3 位偶数](https://leetcode-cn.com/problems/finding-3-digit-even-numbers/)

```python
class Solution:
    def findEvenNumbers(self, digits: List[int]) -> List[int]:
        cnt = collections.Counter(sorted(digits))
        ans = []
        
        def dfs(path):
            if len(path) == 3:
                res = path[0] * 100 + path[1] * 10 + path[2]
                if res > 99 and res % 2 == 0:
                    ans.append(res)
                return
            for i in cnt:
                if cnt[i] > 0:
                    cnt[i] -= 1
                    path.append(i)
                    dfs(path)
                    path.pop()
                    cnt[i] += 1
        dfs([])
        return ans
```
