[316. 去除重复字母](https://leetcode-cn.com/problems/remove-duplicate-letters/)

## 方法：移除问题+单调栈

这道题的母题是 402 题



```c++
class Solution {
public:
    string removeDuplicateLetters(string s) {
        // 当遍历到一个元素时，如果前面出现过这个元素，就跳过
        // 弹出时也要注意，如果后面没有剩的，不能弹出
        // memory 确保了堆内的元素仅会出现一次
        vector<char> stk;
        unordered_set<char> memory;
        unordered_map<char, int> remain;
        for (char c : s) remain[c]++;
        for (char c : s) {
            if (memory.count(c) == 0) {
                while (!stk.empty() && c < *stk.rbegin() && remain[*stk.rbegin()] != 0) {
                    memory.erase(*stk.rbegin());
                    stk.pop_back();
                }
                stk.push_back(c);
                memory.insert(c);
            }
            remain[c]--;
        }
        string ans;
        for (char c : stk) ans += c;
        // cout << ans << endl;
        return ans;
    }
};
```
