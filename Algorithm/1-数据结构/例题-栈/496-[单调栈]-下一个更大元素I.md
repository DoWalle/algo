题目：[496. 下一个更大元素 I](https://leetcode.cn/problems/next-greater-element-i/)

```c++
class Solution {
public:
    vector<int> nextGreaterElement(vector<int>& nums1, vector<int>& nums2) {
        int n1 = nums1.size();
        int n2 = nums2.size();
        unordered_map<int, int> record;
        stack<int> stk;

        for (int i = n2 - 1; i >= 0; i--) {
            while (!stk.empty() && stk.top() <= nums2[i]) {
                stk.pop();
            }
            record[nums2[i]] = stk.empty() ? -1 : stk.top();
            stk.push(nums2[i]);
        }

        vector<int> ans(n1, 0);
        for (int i = 0; i < n1; i++) {
            ans[i] = record[nums1[i]];
        }

        return ans;
    }
};
```
