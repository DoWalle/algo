题目：[面试题 01.05. 一次编辑](https://leetcode.cn/problems/one-away-lcci/)

```python
class Solution:
    def oneEditAway(self, first: str, second: str) -> bool:
        n1 = len(first)
        n2 = len(second)
        idx1 = 0
        idx2 = 0
        cnt = 0  # 更改次数
        flag = True  # 是否相同

        if n1+1 == n2:  # 1、插
            while idx1 < n1 and idx2 < n2:
                if first[idx1] != second[idx2]:
                    if cnt == 1:
                        flag = False
                        break
                    idx2 += 1
                    cnt += 1
                else:
                    idx1 += 1
                    idx2 += 1
        elif n1-1 == n2:  # 2、删
            while idx1 < n1 and idx2 < n2:
                if first[idx1] != second[idx2]:
                    if cnt == 1:
                        flag = False
                        break
                    idx1 += 1
                    cnt += 1
                else:
                    idx1 += 1
                    idx2 += 1
        elif n1 == n2:  # 3、改
            while idx1 < n1 and idx2 < n2:
                if first[idx1] != second[idx2]:
                    if cnt == 1:
                        flag = False
                        break
                    idx1 += 1
                    idx2 += 1
                    cnt += 1
                else:
                    idx1 += 1
                    idx2 += 1
        else:
            return False
        if flag:
            return True
        return False

```
