[190. 颠倒二进制位](https://leetcode-cn.com/problems/reverse-bits/)

```c++
class Solution {
public:
    uint32_t reverseBits(uint32_t n) {
        uint32_t ans = 0;
        // 进制的本质
        int i = 32;
        while (i--) {
            ans <<= 1;     // 先扩充一位
            ans += n & 1;  // 往扩充的那一位上，添加原数字的最后一位
            n >>= 1;       // 舍弃原数字的最后一位
        }
        return ans;
    }
};
```
