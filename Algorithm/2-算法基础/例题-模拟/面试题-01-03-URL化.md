题目：[面试题 01.03. URL化](https://leetcode.cn/problems/string-to-url-lcci/)

```c++
class Solution {
public:
    string replaceSpaces(string S, int length) {
        int i = S.size() - 1;  // 要填充的下标
        int j = length - 1;    // 原字符串下标
        while (j >= 0) {
            if (S[j] == ' ') {
                S[i] = '0';
                i--;
                S[i] = '2';
                i--;
                S[i] = '%';
                i--;
            } else {
                S[i] = S[j];
                i--;
            }
            j--;
        }
        cout << S.size() << endl;
        return S.substr(i + 1);
    }
};
```
