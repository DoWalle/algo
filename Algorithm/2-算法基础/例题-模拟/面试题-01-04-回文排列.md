题目：[面试题 01.04. 回文排列](https://leetcode.cn/problems/palindrome-permutation-lcci/)

```c++
class Solution {
public:
    bool canPermutePalindrome(string s) {
        vector<int> arr(128, 0);
        for (char c : s) {
            arr[c]++;
        }
        int even = 0;
        for (int i = 0; i < 128; i++) {
            if (arr[i] & 1) {
                even++;
            }
        }
        return even > 1 ? false : true;
    }
};
```
